#include<iostream>

using namespace std;

int main()
{
	int tab[2][2];
	int x, y;
	for (x = 0; x < 2; x++)
	{
		for (y = 0; y < 2; y++)
		{
			cin >> tab[x][y];
		}
	}

	cout << "==========" << endl;

	for (x = 0; x < 2; x++)
	{
		for (y = 0; y < 2; y++)
		{
			cout << tab[x][y] << "  ";
		}
		cout << endl;
	}

	cout << "==========" << endl;
}
